nano install_nginx.yml

---
- name: Instalar NGINX en cliente Linux
  hosts: cliente1
  become: true
  tasks:
    - name: Instalar NGINX
      apt:
        name: nginx
        state: present
        update_cache: yes

    - name: Iniciar el servicio NGINX
      service:
        name: nginx
        state: started
        enabled: true



ansible-playbook install_nginx.yml
